%% Modify \section so that it automatically includes a line below

\newcommand{\sectionline}{%
  \vspace{-0.2cm}%
  \hrule%
  \vspace{0.3cm}%
}

\let\xsection=\section
\renewcommand{\section}[1]{%
  \xsection*{#1}%
  \sectionline%
}
